how Guni works in your agent
YOUR AGENT
Navigates to a URL, gets HTML back from the page
GUNI SCANS
8-vector analysis in ~0.001s. Returns ALLOW / CONFIRM / BLOCK
AGENT ACTS
If ALLOW → proceed. If BLOCK → stop. If CONFIRM → flag & log.
Risk scoreDecisionWhat to do in your agent
≥ 70BLOCKStop. Don't execute. Go back or abort the task.
40–69CONFIRMLog the warning. Proceed carefully. Notify user.
< 40ALLOWSafe to proceed with planned action.

Add the safe_goto wrapper
One function wraps every navigation in your agent. Copy this exactly.
python
from guni import GuniScanner
from playwright.sync_api import sync_playwright

# Create scanner once with your agent's goal
scanner = GuniScanner(goal="Book a train ticket")

def safe_goto(page, url: str) -> bool:
    """Navigate safely. Returns False if blocked."""
    page.goto(url)
    result = scanner.scan(html=page.content(), url=url)

    if result["decision"] == "BLOCK":
        print(f"[GUNI] BLOCKED: {url} — Risk {result['risk']}/100")
        for cat, items in result["evidence"].items():
            for item in items:
                print(f"  [{cat}] {item}")
        page.go_back()
        return False

    if result["decision"] == "CONFIRM":
        print(f"[GUNI] WARNING: {url} — Risk {result['risk']}/100")

    return True  # ALLOW — safe to proceed

Wrap your agent with Guni protection
python
from guni import GuniScanner
from browser_use import Agent
from langchain_anthropic import ChatAnthropic
import asyncio

async def run_secure_agent(task: str):
    scanner = GuniScanner(goal=task)
    llm     = ChatAnthropic(model="claude-3-5-sonnet-20241022")
    agent   = Agent(task=task, llm=llm)

    original_step = agent.step

    async def secure_step(*args, **kwargs):
        if hasattr(agent, 'browser') and agent.browser:
            page   = await agent.browser.get_current_page()
            result = scanner.scan(
                html=await page.content(),
                url=page.url
            )
            if result["decision"] == "BLOCK":
                return f"Blocked by Guni: {page.url}"
        return await original_step(*args, **kwargs)

    agent.step = secure_step
    return await agent.run()

asyncio.run(run_secure_agent("Book cheapest flight Delhi to Mumbai"))

Real-time scanning as agent browses
Best for high-frequency agents. One persistent connection, stream pages as they load.
python
import asyncio, json, websockets

async def secure_agent():
    uri = "wss://guni.up.railway.app/ws/scan?goal=Book+a+flight"

    async with websockets.connect(uri) as ws:
        async def check_page(html, url):
            await ws.send(json.dumps({"html": html, "url": url}))
            resp = json.loads(await ws.recv())
            # skip "scanning" status message
            while resp.get("type") == "scanning":
                resp = json.loads(await ws.recv())
            return resp

        # In your agent loop:
        result = await check_page(page_html, current_url)
        if result.get("decision") == "BLOCK":
            print(f"BLOCKED: {current_url}")
            return

asyncio.run(secure_agent())
